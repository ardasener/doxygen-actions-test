449inline std::string unescape(const std::string& codepoint)
450{
451 std::uint32_t x;
452 std::uint8_t buf[8];
453 std::stringstream ss(codepoint);
454
455 ss >> std::hex >> x;
456
457 if (x <= 0x7FUL) {
458 // 0xxxxxxx
459 buf[0] = 0x00 | ((x >> 0) & 0x7F);
460 buf[1] = '\0';
461 } else if (x <= 0x7FFUL) {
462 // 110yyyyx 10xxxxxx
463 buf[0] = 0xC0 | ((x >> 6) & 0xDF);
464 buf[1] = 0x80 | ((x >> 0) & 0xBF);
465 buf[2] = '\0';
466 } else if (x <= 0xFFFFUL) {
467 // 1110yyyy 10yxxxxx 10xxxxxx
468 buf[0] = 0xE0 | ((x >> 12) & 0xEF);
469 buf[1] = 0x80 | ((x >> 6) & 0xBF);
470 buf[2] = 0x80 | ((x >> 0) & 0xBF);
471 buf[3] = '\0';
472 } else if (x <= 0x10FFFFUL) {
473 // 11110yyy 10yyxxxx 10xxxxxx 10xxxxxx
474 buf[0] = 0xF0 | ((x >> 18) & 0xF7);
475 buf[1] = 0x80 | ((x >> 12) & 0xBF);
476 buf[2] = 0x80 | ((x >> 6) & 0xBF);
477 buf[3] = 0x80 | ((x >> 0) & 0xBF);
478 buf[4] = '\0';
479 } else {
480 buf[0] = '\0';
481 }
482
483 return reinterpret_cast<char*>(buf);
484}

486// Returns true if |s| is integer.
487// [+-]?\d+(_\d+)*
488inline bool isInteger(const std::string& s)
489{
490 if (s.empty())
491 return false;
492
493 std::string::size_type p = 0;
494 if (s[p] == '+' || s[p] == '-')
495 ++p;
496
497 while (p < s.size() && '0' <= s[p] && s[p] <= '9') {
498 ++p;
499 if (p < s.size() && s[p] == '_') {
500 ++p;
501 if (!(p < s.size() && '0' <= s[p] && s[p] <= '9'))
502 return false;
503 }
504 }
505
506 return p == s.size();
507}
508
509// Returns true if |s| is double.
510// [+-]? (\d+(_\d+)*)? (\.\d+(_\d+)*)? ([eE] [+-]? \d+(_\d+)*)?
511// 1----------- 2------------- 3----------------------
512// 2 or (1 and 3) should exist.
513inline bool isDouble(const std::string& s)
514{
515 if (s.empty())
516 return false;
517
518 std::string::size_type p = 0;
519 if (s[p] == '+' || s[p] == '-')
520 ++p;
521
522 bool ok = false;
523 while (p < s.size() && '0' <= s[p] && s[p] <= '9') {
524 ++p;
525 ok = true;
526
527 if (p < s.size() && s[p] == '_') {
528 ++p;
529 if (!(p < s.size() && '0' <= s[p] && s[p] <= '9'))
530 return false;
531 }
532 }
533
534 if (p < s.size() && s[p] == '.')
535 ++p;
536
537 while (p < s.size() && '0' <= s[p] && s[p] <= '9') {
538 ++p;
539 ok = true;
540
541 if (p < s.size() && s[p] == '_') {
542 ++p;
543 if (!(p < s.size() && '0' <= s[p] && s[p] <= '9'))
544 return false;
545 }
546 }
547
548 if (!ok)
549 return false;
550
551 ok = false;
552 if (p < s.size() && (s[p] == 'e' || s[p] == 'E')) {
553 ++p;
554 if (p < s.size() && (s[p] == '+' || s[p] == '-'))
555 ++p;
556 while (p < s.size() && '0' <= s[p] && s[p] <= '9') {
557 ++p;
558 ok = true;
559
560 if (p < s.size() && s[p] == '_') {
561 ++p;
562 if (!(p < s.size() && '0' <= s[p] && s[p] <= '9'))
563 return false;
564 }
565 }
566 if (!ok)
567 return false;
568 }
569
570 return p == s.size();
571}

661inline Token Lexer::nextStringDoubleQuote()
662{
663 if (!consume('"'))
664 return Token(TokenType::ERROR_TOKEN, std::string("string didn't start with '\"'"));
665
666 std::string s;
667 char c;
668 bool multiline = false;
669
670 if (current(&c) && c == '"') {
671 next();
672 if (!current(&c) || c != '"') {
673 // OK. It's empty string.
674 return Token(TokenType::STRING, std::string());
675 }
676
677 next();
678 // raw string literal started.
679 // Newline just after """ should be ignored.
680 while (current(&c) && (c == ' ' || c == '\t'))
681 next();
682 if (current(&c) && c == '\n')
683 next();
684 multiline = true;
685 }
686
687 while (current(&c)) {
688 next();
689 if (c == '\\') {
690 if (!current(&c))
691 return Token(TokenType::ERROR_TOKEN, std::string("string has unknown escape sequence"));
692 next();
693 switch (c) {
694 case 't': c = '\t'; break;
695 case 'n': c = '\n'; break;
696 case 'r': c = '\r'; break;
697 case 'u':
698 case 'U': {
699 int size = c == 'u' ? 4 : 8;
700 std::string codepoint;
701 for (int i = 0; i < size; ++i) {
702 if (current(&c) && (('0' <= c && c <= '9') || ('A' <= c && c <= 'F') || ('a' <= c && c <= 'f'))) {
703 codepoint += c;
704 next();
705 } else {
706 return Token(TokenType::ERROR_TOKEN, std::string("string has unknown escape sequence"));
707 }
708 }
709 s += unescape(codepoint);
710 continue;
711 }
712 case '"': c = '"'; break;
713 case '\'': c = '\''; break;
714 case '\\': c = '\\'; break;
715 case '\n':
716 while (current(&c) && (c == ' ' || c == '\t' || c == '\r' || c == '\n')) {
717 next();
718 }
719 continue;
720 default:
721 return Token(TokenType::ERROR_TOKEN, std::string("string has unknown escape sequence"));
722 }
723 } else if (c == '"') {
724 if (multiline) {
725 if (current(&c) && c == '"') {
726 next();
727 if (current(&c) && c == '"') {
728 next();
729 return Token(TokenType::MULTILINE_STRING, s);
730 } else {
731 s += '"';
732 s += '"';
733 continue;
734 }
735 } else {
736 s += '"';
737 continue;
738 }
739 } else {
740 return Token(TokenType::STRING, s);
741 }
742 }
743
744 s += c;
745 }
746
747 return Token(TokenType::ERROR_TOKEN, std::string("string didn't end"));
748}

813 while (current(&c) && (isalnum(static_cast<unsigned char>(c)) || c == '_' || c == '-')) {
814 s += c;
815 next();
816 }
817
818 if (s.empty())
819 return Token(TokenType::ERROR_TOKEN, std::string("Unknown key format"));
820
821 return Token(TokenType::IDENT, s);
822}

824inline Token Lexer::nextValue()
825{
826 std::string s;
827 char c;
828
829 if (current(&c) && isalpha(static_cast<unsigned char>(c))) {
830 s += c;
831 next();
832 while (current(&c) && isalpha(static_cast<unsigned char>(c))) {
833 s += c;
834 next();
835 }
836
837 if (s == "true")
838 return Token(TokenType::BOOL, true);
839 if (s == "false")
840 return Token(TokenType::BOOL, false);
841 return Token(TokenType::ERROR_TOKEN, std::string("Unknown ident: ") + s);
842 }
843
844 while (current(&c) && (('0' <= c && c <= '9') || c == '.' || c == 'e' || c == 'E' ||
845 c == 'T' || c == 'Z' || c == '_' || c == ':' || c == '-' || c == '+')) {
846 next();
847 s += c;
848 }
849
850 if (isInteger(s)) {
851 std::stringstream ss(removeDelimiter(s));
852 std::int64_t x;
853 ss >> x;
854 return Token(TokenType::INT, x);
855 }
856
857 if (isDouble(s)) {
858 std::stringstream ss(removeDelimiter(s));
859 double d;
860 ss >> d;
861 return Token(TokenType::DOUBLE, d);
862 }
863
864 return parseAsTime(s);
865}

867inline Token Lexer::parseAsTime(const std::string& str)
868{
869 const char* s = str.c_str();
870
871 int n;
872 int YYYY, MM, DD;
873#if defined(_MSC_VER)
874 if (sscanf_s(s, "%d-%d-%d%n", &YYYY, &MM, &DD, &n) != 3)
875 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
876#else
877 if (sscanf(s, "%d-%d-%d%n", &YYYY, &MM, &DD, &n) != 3)
878 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
879#endif
880
881 if (!(1 <= MM && MM <= 12)) {
882 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
883 }
884 if (YYYY < 1900) {
885 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
886 }
887
888 if (s[n] == '\0') {
889 std::tm t;
890 t.tm_sec = 0;
891 t.tm_min = 0;
892 t.tm_hour = 0;
893 t.tm_mday = DD;
894 t.tm_mon = MM - 1;
895 t.tm_year = YYYY - 1900;
896 auto tp = std::chrono::system_clock::from_time_t(timegm(&t));
897 return Token(TokenType::TIME, tp);
898 }
899
900 if (s[n] != 'T')
901 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
902
903 s = s + n + 1;
904
905 int hh, mm;
906 double ss; // double for fraction
907#if defined(_MSC_VER)
908 if (sscanf_s(s, "%d:%d:%lf%n", &hh, &mm, &ss, &n) != 3)
909 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
910#else
911 if (sscanf(s, "%d:%d:%lf%n", &hh, &mm, &ss, &n) != 3)
912 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
913#endif
914
915 std::tm t;
916 t.tm_sec = static_cast<int>(ss);
917 t.tm_min = mm;
918 t.tm_hour = hh;
919 t.tm_mday = DD;
920 t.tm_mon = MM - 1;
921 t.tm_year = YYYY - 1900;
922 auto tp = std::chrono::system_clock::from_time_t(timegm(&t));
923 ss -= static_cast<int>(ss);
924 // TODO(mayah): workaround GCC 4.9.3 on cygwin does not have std::round, but round().
925 tp += std::chrono::microseconds(static_cast<std::int64_t>(round(ss * 1000000)));
926
927 if (s[n] == '\0')
928 return Token(TokenType::TIME, tp);
929
930 if (s[n] == 'Z' && s[n + 1] == '\0')
931 return Token(TokenType::TIME, tp);
932
933 s = s + n;
934 // offset
935 // [+/-]%d:%d
936 char pn;
937 int oh, om;
938#if defined(_MSC_VER)
939 if (sscanf_s(s, "%c%d:%d", &pn, static_cast<unsigned>(sizeof(pn)), &oh, &om) != 3)
940 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
941#else
942 if (sscanf(s, "%c%d:%d", &pn, &oh, &om) != 3)
943 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
944#endif
945
946 if (pn != '+' && pn != '-')
947 return Token(TokenType::ERROR_TOKEN, std::string("Invalid token"));
948
949 if (pn == '+') {
950 tp -= std::chrono::hours(oh);
951 tp -= std::chrono::minutes(om);
952 } else {
953 tp += std::chrono::hours(oh);
954 tp += std::chrono::minutes(om);
955 }
956
957 return Token(TokenType::TIME, tp);
958}

970inline Token Lexer::nextToken(bool isValueToken)
971{
972 char c;
973 while (current(&c)) {
974 if (c == ' ' || c == '\t' || c == '\r') {
975 next();
976 continue;
977 }
978
979 if (c == '#') {
980 skipUntilNewLine();
981 continue;
982 }
983
984 switch (c) {
985 case '\n':
986 next();
987 return Token(TokenType::END_OF_LINE);
988 case '=':
989 next();
990 return Token(TokenType::EQUAL);
991 case '{':
992 next();
993 return Token(TokenType::LBRACE);
994 case '}':
995 next();
996 return Token(TokenType::RBRACE);
997 case '[':
998 next();
999 return Token(TokenType::LBRACKET);
1000 case ']':
1001 next();
1002 return Token(TokenType::RBRACKET);
1003 case ',':
1004 next();
1005 return Token(TokenType::COMMA);
1006 case '.':
1007 next();
1008 return Token(TokenType::DOT);
1009 case '\"':
1010 return nextStringDoubleQuote();
1011 case '\'':
1012 return nextStringSingleQuote();
1013 default:
1014 if (isValueToken) {
1015 return nextValue();
1016 } else {
1017 return nextKey();
1018 }
1019 }
1020 }
1021
1022 return Token(TokenType::END_OF_FILE);
1023}

1332inline void Value::write(std::ostream* os, const std::string& keyPrefix, int indent) const
1333{
1334 switch (type_) {
1335 case NULL_TYPE:
1336 failwith("null type value is not a valid value");
1337 break;
1338 case BOOL_TYPE:
1339 (*os) << (bool_ ? "true" : "false");
1340 break;
1341 case INT_TYPE:
1342 (*os) << int_;
1343 break;
1344 case DOUBLE_TYPE: {
1345 (*os) << std::fixed << std::showpoint << double_;
1346 break;
1347 }
1348 case STRING_TYPE:
1349 (*os) << '"' << internal::escapeString(*string_) << '"';
1350 break;
1351 case TIME_TYPE: {
1352 time_t tt = std::chrono::system_clock::to_time_t(*time_);
1353 std::tm t;
1354 gmtime_r(&tt, &t);
1355 char buf[256];
1356 snprintf(buf, sizeof(buf), "%04d-%02d-%02dT%02d:%02d:%02dZ", t.tm_year + 1900, t.tm_mon + 1, t.tm_mday, t.tm_hour, t.tm_min, t.tm_sec);
1357 (*os) << buf;
1358 break;
1359 }
1360 case ARRAY_TYPE:
1361 (*os) << '[';
1362 for (size_t i = 0; i < array_->size(); ++i) {
1363 if (i)
1364 (*os) << ", ";
1365 (*array_)[i].write(os, keyPrefix, -1);
1366 }
1367 (*os) << ']';
1368 break;
1369 case TABLE_TYPE:
1370 for (const auto& kv : *table_) {
1371 if (kv.second.is<Table>())
1372 continue;
1373 if (kv.second.is<Array>() && kv.second.size() > 0 && kv.second.find(0)->is<Table>())
1374 continue;
1375 (*os) << spaces(indent) << escapeKey(kv.first) << " = ";
1376 kv.second.write(os, keyPrefix, indent >= 0 ? indent + 1 : indent);
1377 (*os) << '\n';
1378 }
1379 for (const auto& kv : *table_) {
1380 if (kv.second.is<Table>()) {
1381 std::string key(keyPrefix);
1382 if (!keyPrefix.empty())
1383 key += ".";
1384 key += escapeKey(kv.first);
1385 (*os) << "\n" << spaces(indent) << "[" << key << "]\n";
1386 kv.second.write(os, key, indent >= 0 ? indent + 1 : indent);
1387 }
1388 if (kv.second.is<Array>() && kv.second.size() > 0 && kv.second.find(0)->is<Table>()) {
1389 std::string key(keyPrefix);
1390 if (!keyPrefix.empty())
1391 key += ".";
1392 key += escapeKey(kv.first);
1393 for (const auto& v : kv.second.as<Array>()) {
1394 (*os) << "\n" << spaces(indent) << "[[" << key << "]]\n";
1395 v.write(os, key, indent >= 0 ? indent + 1 : indent);
1396 }
1397 }
1398 }
1399 break;
1400 default:
1401 failwith("writing unknown type");
1402 break;
1403 }
1404}
